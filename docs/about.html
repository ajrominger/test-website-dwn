About this site

Download the index.qmd file via github link:

baseURL <- "https://raw.githubusercontent.com/ajrominger/test-website-dwn/refs/heads/main/"
downloadthis::download_link(link = paste0(baseURL, "index.qmd"),
                            button_label = "Download assignment",
                            button_type = "danger",
                            has_icon = TRUE,
                            icon = "fa fa-save",
                            self_contained = FALSE)

Download file

downloadthis::download_file(path = "index.qmd",
                            output_name = "assignment",
                            button_label = "Download assingment",
                            button_type = "warning",
                            has_icon = TRUE,
                            icon = "fa fa-save",
                            self_contained = FALSE)

Download via temp file

x <- readLines("index.qmd") |> 
    c("\n", "via temp \n")
d <- tempdir()

writeLines(x, file.path(d, "tmp.qmd"))

downloadthis::download_file(path = file.path(d, "tmp.qmd"),
                            output_name = "assignment",
                            button_label = "Download assingment",
                            button_type = "primary",
                            has_icon = TRUE,
                            icon = "fa fa-save",
                            self_contained = FALSE)
